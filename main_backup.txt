#日志模块 
def console_out(logFilename):  
    ''''' Output log to file and console '''  
    # Define a Handler and set a format which output to file  
    logging.basicConfig(  
                    level    = logging.INFO,                                                                         
                    format   = '[%(asctime)s]  [%(filename)s] : [%(levelname)s]  %(message)s',   
                    datefmt  = '%Y-%m-%d %A %H:%M:%S',                                     
                    filename = logFilename,             
                    filemode = 'w')                    
    # Define a Handler and set a format which output to console  
    console = logging.StreamHandler()          
    console.setLevel(logging.INFO)        
    formatter = logging.Formatter('%(asctime)s : %(levelname)s  %(message)s')  
    console.setFormatter(formatter)  
    # Create an instance  
    logging.getLogger().addHandler(console)

class Send_Email:
    def __init__(cls,file_path):
        cls.file_path=file_path
        cls.from_address="jenkinsrobot@sohu.com"
        cls.to_address=["shifawu@tuniu.com"]
    def new_file(cls):
        list=os.listdir(cls.file_path)
        list.sort(key=lambda fn:os.path.getmtime(cls.file_path+"\\"+fn))
        cls.file_new1=os.path.join(cls.file_path,list[-1])
    def send_mail(cls,contents=""):

        #附件Logging文件
        list=os.listdir(cls.file_path)
        list.sort(key=lambda fn:os.path.getmtime(cls.file_path+"\\"+fn))
        for i in range(len(list)):
            if list[i].endswith("log"):
                log_name=list[i]
        cls.file_log=os.path.join(cls.file_path,log_name)
        #log_path1=os.path.join(cls.file_path,cls.file_log)

        for i in range(len(list)):
            if "report" in list[i]:
                report_name=list[i]
        cls.file_new=os.path.join(cls.file_path,report_name)

        with open(cls.file_new,"rb") as f:
            mail_body=f.read()
        msg=MIMEMultipart() 
        msg['Subject']=u"APP端UI主流程自动化测试报告"
        #msg.attach(MIMEText(mail_body,"html","utf-8"))
        msg['From'] =cls.from_address      # 发件人
        msg['To'] = ','.join(cls.to_address)
        with open(cls.file_log,'rb') as f:
            log_body=f.read()
        att1=MIMEText(log_body,"base64","")
        #att1.set_payload(log_body)
        #encode_base64(att1)
        #att1.add_header('Content-Disposition', 'attachment',   filename="logging.log") 
        #att1["Content-Type"] = 'application/octet-stream'
        att1["Content-Disposition"] = 'attachment; filename="Logging.log"'
        msg.attach(att1)
        if not contents:
            contents=mail_body
            msg.attach(MIMEText(contents,"html","gbk"))
        else:
            msg.attach(MIMEText(contents,"html","gbk"))

        smtp=smtplib.SMTP()
        smtp.connect('smtp.sohu.com')
        smtp.login(cls.from_address,"shi@123456")
        smtp.sendmail(cls.from_address,cls.to_address,msg.as_string())
        #print "this is mail==================================",mail_body
        smtp.quit()
